# Taking a look at number of movies in the top 1000 for each year

unique_years = movie_data.Released_Year.unique()
unique_year_amounts = []
for year in movie_data.Released_Year.unique():
    current_year_amount = len(movie_data[movie_data["Released_Year"] == year].index)
    unique_year_amounts.append(current_year_amount)
    
fig, ax1 = plt.subplots(figsize=(20,12))
ax1.bar(unique_years, unique_year_amounts)
ax1.set_xticks(np.arange(int(min(unique_years)), int(max(unique_years)), 5))
ax1.set_xlabel("Year")
ax1.set_ylabel("Number of Movies")
ax1.set_title("Bar Plot of Number of Movies in the Top 1000 based on IMDB Ratings by Year")
